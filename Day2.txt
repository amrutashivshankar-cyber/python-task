# String Modification Functions  
s = "Python is a High level programming Language"  
print("Lowercase:", s.lower())  # python is a high level programming language  
print("Uppercase:", s.upper())  # PYTHON IS A HIGH LEVEL PROGRAMMING LANGUAGE  
print("Swapcase:", s.swapcase())  # pYTHON IS A hIGH LEVEL PROGRAMMING lANGUAGE  
print("Title Case:", s.title())  # Python Is A High Level Programming Language  
print("Capitalized:", s.capitalize())  # Python is a high level programming language  
  
# String Slicing Operations  
name = "prashantjha"  # Indexing: 012345678910  
print("First character:", name[0])  # p  
print("Second character:", name[1])  # r  
print("Last character:", name[-1])  # a  
  
# Slicing examples  
print("Substring [0:5]:", name[0:5])  # prash  
print("Substring [1:]:", name[1:])  # rashantjha  
print("Substring [:5]:", name[:5])  # prash  
print("Full string [:]:", name[:])  # prashantjha  
print("Substring with step [1:8:2]:", name[1:8:2])  # rsat  
print("Reversed string:", name[::-1])  # ahjtnahsarp
